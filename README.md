# CREAR BASE DE DATOS:
CREATE DATABASE alwaysmusic;

# CREAR LA TABLA:
CREATE TABLE estudiantes(
	id serial PRIMARY KEY NOT NULL,
	nombre varchar(255) NOT NULL,
	rut varchar(13) NOT NULL UNIQUE,
	curso varchar(50) NOT NULL,
	nivel int NOT NULL,
);
